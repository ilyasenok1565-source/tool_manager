<script>
    // Глобальные переменные
    let currentMode = null;
    let codeReader = null;
    let scanning = false;
    let currentUser = null;
    let tempToolQR = null;

    // DOM элементы
    const loginContainer = document.getElementById('loginContainer');
    const appContainer = document.getElementById('app');
    const loginBtn = document.getElementById('loginBtn');
    const logoutBtn = document.getElementById('logoutBtn');
    const usernameInput = document.getElementById('username');
    const passwordInput = document.getElementById('password');
    const loginError = document.getElementById('loginError');
    const issuedListDiv = document.getElementById('issuedList');
    const btnIssue = document.getElementById('btnIssue');
    const btnReturn = document.getElementById('btnReturn');
    const scannerArea = document.getElementById('scannerArea');
    const startScannerBtn = document.getElementById('startScanner');
    const stopScannerBtn = document.getElementById('stopScanner');
    const video = document.getElementById('video');
    const scanResultDiv = document.getElementById('scanResult');
    const infoArea = document.getElementById('infoArea');
    const tabs = document.querySelectorAll('.tab');
    const tabContents = document.querySelectorAll('.tab-content');
    const themeToggle = document.getElementById('themeToggle');

    // Админские элементы
    const addToolBtn = document.getElementById('addToolBtn');
    const addEmpBtn = document.getElementById('addEmpBtn');
    const addUserBtn = document.getElementById('addUserBtn');

    const employeesTableBody = document.querySelector('#employeesTable tbody');
    const usersTableBody = document.querySelector('#usersTable tbody');

    // Тема
    function initTheme() {
        const savedTheme = localStorage.getItem('theme');
        if (savedTheme === 'dark') {
            document.body.classList.add('dark');
            themeToggle.innerHTML = '<i class="fas fa-sun"></i>';
        } else {
            document.body.classList.remove('dark');
            themeToggle.innerHTML = '<i class="fas fa-moon"></i>';
        }
    }
    function toggleTheme() {
        if (document.body.classList.contains('dark')) {
            document.body.classList.remove('dark');
            localStorage.setItem('theme', 'light');
            themeToggle.innerHTML = '<i class="fas fa-moon"></i>';
        } else {
            document.body.classList.add('dark');
            localStorage.setItem('theme', 'dark');
            themeToggle.innerHTML = '<i class="fas fa-sun"></i>';
        }
    }
    themeToggle.onclick = toggleTheme;

    // Авторизация
    async function checkAuth() {
        try {
            const resp = await fetch('/auth/me');
            if (resp.ok) {
                currentUser = await resp.json();
                loginContainer.style.display = 'none';
                appContainer.style.display = 'block';
                if (currentUser.role === 'admin') {
                    document.querySelector('.admin-tab').style.display = 'block';
                } else {
                    document.querySelector('.admin-tab').style.display = 'none';
                }
                loadIssued();
                if (currentUser.role === 'admin') {
                    loadTools();
                    loadEmployees();
                    loadUsers();
                }
                return true;
            } else {
                loginContainer.style.display = 'block';
                appContainer.style.display = 'none';
                return false;
            }
        } catch (err) {
            loginContainer.style.display = 'block';
            appContainer.style.display = 'none';
            return false;
        }
    }

    async function login() {
        const username = usernameInput.value.trim();
        const password = passwordInput.value;
        if (!username || !password) {
            loginError.textContent = 'Введите логин и пароль';
            return;
        }
        loginError.textContent = '';
        const formData = new URLSearchParams();
        formData.append('username', username);
        formData.append('password', password);
        try {
            const resp = await fetch('/auth/login', {
                method: 'POST',
                headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
                body: formData
            });
            if (resp.ok) {
                await checkAuth();
            } else {
                const err = await resp.json();
                loginError.textContent = err.detail || 'Ошибка входа';
            }
        } catch (err) {
            loginError.textContent = 'Ошибка соединения';
        }
    }

    async function logout() {
        await fetch('/auth/logout', { method: 'POST' });
        currentUser = null;
        loginContainer.style.display = 'block';
        appContainer.style.display = 'none';
        if (codeReader && scanning) codeReader.reset();
    }

    // Загрузка данных
    async function loadIssued() {
        try {
            const resp = await fetch('/issued');
            if (!resp.ok) throw new Error();
            const data = await resp.json();
            if (data.length === 0) {
                issuedListDiv.innerHTML = 'Нет выданных инструментов';
                return;
            }
            let html = '';
            for (let item of data) {
                html += `<div class="tool-item"><i class="fas fa-toolbox"></i> <strong>${item.tool_name}</strong><br>
                        <i class="fas fa-user"></i> Выдан: ${item.employee_name} (таб. ${item.tab_number})<br>
                        <i class="fas fa-archive"></i> Контейнер: ${item.container || 'не указан'}</div>`;
            }
            issuedListDiv.innerHTML = html;
        } catch (err) {
            issuedListDiv.innerHTML = 'Ошибка загрузки';
        }
    }

    async function loadTools() {
        const resp = await fetch('/admin/tools');
        const tools = await resp.json();
        toolsTableBody.innerHTML = '';
        for (let tool of tools) {
            const row = toolsTableBody.insertRow();
            row.insertCell(0).textContent = tool.id;
            row.insertCell(1).textContent = tool.name;
            row.insertCell(2).textContent = tool.status === 'issued' ? 'Выдан' : 'На складе';
            row.insertCell(3).textContent = tool.container || '';
            row.insertCell(4).innerHTML = `<a href="/qrcodes/${tool.qr_code}.png" target="_blank"><i class="fas fa-qrcode"></i> QR</a>`;
            const actions = row.insertCell(5);
            const delBtn = document.createElement('button');
            delBtn.innerHTML = '<i class="fas fa-trash"></i>';
            delBtn.className = 'danger';
            delBtn.onclick = () => deleteTool(tool.id);
            actions.appendChild(delBtn);
        }
    }

    async function deleteTool(id) {
        if (!confirm('Удалить инструмент?')) return;
        const resp = await fetch(`/admin/tools/${id}`, { method: 'DELETE' });
        if (resp.ok) loadTools();
        else alert((await resp.json()).detail);
    }

    async function addTool() {
        const name = document.getElementById('newToolName').value.trim();
        const container = document.getElementById('newToolContainer').value.trim();
        if (!name) return alert('Введите название');
        const resp = await fetch('/admin/tools', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ name, container })
        });
        if (resp.ok) {
            document.getElementById('newToolName').value = '';
            document.getElementById('newToolContainer').value = '';
            loadTools();
        } else alert((await resp.json()).detail);
    }

    async function loadEmployees() {
        const resp = await fetch('/admin/employees');
        const employees = await resp.json();
        employeesTableBody.innerHTML = '';
        for (let emp of employees) {
            const row = employeesTableBody.insertRow();
            row.insertCell(0).textContent = emp.id;
            row.insertCell(1).textContent = emp.name;
            row.insertCell(2).textContent = emp.tab_number;
            row.insertCell(3).innerHTML = `<a href="/qrcodes/${emp.qr_code}.png" target="_blank"><i class="fas fa-qrcode"></i> QR</a>`;
            const actions = row.insertCell(4);
            const delBtn = document.createElement('button');
            delBtn.innerHTML = '<i class="fas fa-trash"></i>';
            delBtn.className = 'danger';
            delBtn.onclick = () => deleteEmployee(emp.id);
            actions.appendChild(delBtn);
        }
    }

    async function deleteEmployee(id) {
        if (!confirm('Удалить сотрудника?')) return;
        const resp = await fetch(`/admin/employees/${id}`, { method: 'DELETE' });
        if (resp.ok) loadEmployees();
        else alert((await resp.json()).detail);
    }

    async function addEmployee() {
        const name = document.getElementById('newEmpName').value.trim();
        const tab = document.getElementById('newEmpTab').value.trim();
        if (!name || !tab) return alert('Заполните все поля');
        const resp = await fetch('/admin/employees', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ name, tab_number: tab })
        });
        if (resp.ok) {
            document.getElementById('newEmpName').value = '';
            document.getElementById('newEmpTab').value = '';
            loadEmployees();
        } else alert((await resp.json()).detail);
    }

    async function loadUsers() {
        const resp = await fetch('/admin/users');
        const users = await resp.json();
        usersTableBody.innerHTML = '';
        for (let user of users) {
            const row = usersTableBody.insertRow();
            row.insertCell(0).textContent = user.id;
            row.insertCell(1).textContent = user.username;
            row.insertCell(2).textContent = user.role === 'admin' ? 'Администратор' : 'Кладовщик';
            const actions = row.insertCell(3);
            const roleSelect = document.createElement('select');
            roleSelect.innerHTML = '<option value="worker">Кладовщик</option><option value="admin">Администратор</option>';
            roleSelect.value = user.role;
            roleSelect.onchange = () => updateUserRole(user.id, roleSelect.value);
            actions.appendChild(roleSelect);
            const delBtn = document.createElement('button');
            delBtn.innerHTML = '<i class="fas fa-trash"></i>';
            delBtn.className = 'danger';
            delBtn.onclick = () => deleteUser(user.id);
            actions.appendChild(delBtn);
        }
    }

    async function updateUserRole(userId, newRole) {
        const resp = await fetch(`/admin/users/${userId}/role`, {
            method: 'PUT',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ role: newRole })
        });
        if (resp.ok) loadUsers();
        else alert((await resp.json()).detail);
    }

    async function deleteUser(userId) {
        if (!confirm('Удалить пользователя?')) return;
        const resp = await fetch(`/admin/users/${userId}`, { method: 'DELETE' });
        if (resp.ok) loadUsers();
        else alert((await resp.json()).detail);
    }

    async function addUser() {
        const username = document.getElementById('newUserLogin').value.trim();
        const password = document.getElementById('newUserPass').value;
        const role = document.getElementById('newUserRole').value;
        if (!username || !password) return alert('Заполните логин и пароль');
        const resp = await fetch('/admin/users', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ username, password, role })
        });
        if (resp.ok) {
            document.getElementById('newUserLogin').value = '';
            document.getElementById('newUserPass').value = '';
            loadUsers();
        } else alert((await resp.json()).detail);
    }

    // Сканер
    function stopScanner() {
        if (codeReader && scanning) {
            codeReader.reset();
            scanning = false;
        }
        video.style.display = 'none';
        startScannerBtn.style.display = 'inline-block';
        stopScannerBtn.style.display = 'none';
        scanResultDiv.innerHTML = '';
        scannerArea.style.display = 'none';
    }

    function startScanner() {
        if (!codeReader) codeReader = new ZXing.BrowserMultiFormatReader();
        codeReader.listVideoInputDevices()
            .then(videoInputDevices => {
                if (videoInputDevices.length === 0) {
                    showInfo('Не найдена камера', true);
                    return;
                }
                const firstDeviceId = videoInputDevices[0].deviceId;
                video.style.display = 'block';
                codeReader.decodeFromVideoDevice(firstDeviceId, video, (result, err) => {
                    if (result) {
                        const text = result.getText();
                        scanResultDiv.innerHTML = `<i class="fas fa-check-circle"></i> Отсканировано: ${text}`;
                        handleScan(text);
                        stopScanner();
                    }
                    if (err && !(err instanceof ZXing.NotFoundException)) console.error(err);
                });
                scanning = true;
                startScannerBtn.style.display = 'none';
                stopScannerBtn.style.display = 'inline-block';
            })
            .catch(err => {
                console.error(err);
                showInfo('Ошибка доступа к камере', true);
            });
    }

    async function handleScan(scannedText) {
        if (currentMode === 'issue') {
            if (!tempToolQR) {
                tempToolQR = scannedText;
                scanResultDiv.innerHTML += `<br>📦 Инструмент принят. Теперь отсканируйте сотрудника.`;
            } else {
                const toolQR = tempToolQR;
                const employeeQR = scannedText;
                try {
                    const resp = await fetch('/issue', {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ tool_qr: toolQR, employee_qr: employeeQR })
                    });
                    const data = await resp.json();
                    if (resp.ok) {
                        showInfo(data.message);
                        loadIssued();
                        if (currentUser.role === 'admin') loadTools();
                    } else {
                        showInfo(data.detail || 'Ошибка', true);
                    }
                } catch (err) {
                    showInfo('Ошибка сервера', true);
                }
                tempToolQR = null;
                currentMode = null;
                scannerArea.style.display = 'none';
            }
        } else if (currentMode === 'return') {
            try {
                const resp = await fetch('/return', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ tool_qr: scannedText })
                });
                const data = await resp.json();
                if (resp.ok) {
                    showInfo(data.message);
                    loadIssued();
                    if (currentUser.role === 'admin') loadTools();
                } else {
                    showInfo(data.detail || 'Ошибка', true);
                }
            } catch (err) {
                showInfo('Ошибка сервера', true);
            }
            currentMode = null;
            scannerArea.style.display = 'none';
        }
    }

    function showInfo(message, isError = false) {
        infoArea.style.display = 'block';
        infoArea.style.backgroundColor = isError ? '#f8d7da' : '#d4edda';
        infoArea.style.color = isError ? '#721c24' : '#155724';
        infoArea.innerHTML = `<i class="fas ${isError ? 'fa-exclamation-triangle' : 'fa-info-circle'}"></i> ${message}`;
        setTimeout(() => { infoArea.style.display = 'none'; }, 3000);
    }

    // Переключение вкладок
    function switchTab(tabId) {
        tabContents.forEach(content => content.classList.remove('active'));
        document.getElementById(`tab-${tabId}`).classList.add('active');
        tabs.forEach(t => t.classList.remove('active'));
        document.querySelector(`.tab[data-tab="${tabId}"]`).classList.add('active');
        if (tabId === 'tools' && currentUser?.role === 'admin') loadTools();
        if (tabId === 'employees' && currentUser?.role === 'admin') loadEmployees();
        if (tabId === 'users' && currentUser?.role === 'admin') loadUsers();
    }

    // Обработчики
    btnIssue.onclick = () => {
        currentMode = 'issue';
        tempToolQR = null;
        scannerArea.style.display = 'block';
        startScanner();
    };
    btnReturn.onclick = () => {
        currentMode = 'return';
        scannerArea.style.display = 'block';
        startScanner();
    };
    startScannerBtn.onclick = startScanner;
    stopScannerBtn.onclick = stopScanner;
    loginBtn.onclick = login;
    logoutBtn.onclick = logout;
    if (addToolBtn) addToolBtn.onclick = addTool;
    if (addEmpBtn) addEmpBtn.onclick = addEmployee;
    if (addUserBtn) addUserBtn.onclick = addUser;

    tabs.forEach(tab => {
        tab.addEventListener('click', () => {
            const tabId = tab.getAttribute('data-tab');
            if (tabId === 'users' && currentUser?.role !== 'admin') return;
            switchTab(tabId);
        });
    });

    // Автообновление
    setInterval(() => {
        if (appContainer.style.display !== 'none') {
            loadIssued();
            if (currentUser?.role === 'admin') {
                const activeTab = document.querySelector('.tab-content.active').id;
                if (activeTab === 'tab-tools') loadTools();
                if (activeTab === 'tab-employees') loadEmployees();
                if (activeTab === 'tab-users') loadUsers();
            }
        }
    }, 5000);

    initTheme();
    checkAuth();
</script>
